59 int WiFiClient::connect(IPAddress ip, uint16_t port)
60 {
61  _sock = getFirstSocket();
62  if (_sock != NO_SOCKET_AVAIL) {
63  Control2WiFi::handleEvents();
64 
65  Control2WiFi::gotResponse = false;
66  Control2WiFi::responseType = NONE;
67 
68  if(!SocketDrv::startClient(uint32_t(ip), port, _sock)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
69  // launch interrupt management function, then try to send request again
70  uint32_t start = millis();
71  while(millis() - start < 10)
72  Control2WiFi::handleEvents();
73  if(!SocketDrv::startClient(uint32_t(ip), port, _sock)) // exit if another error occurs
74  return 0;
75  }
76 
77  // Poll flags until we got a response or timeout occurs
78  uint32_t start = millis();
79  while(((millis() - start) < GENERAL_TIMEOUT)){
80  Control2WiFi::handleEvents();
81  if(Control2WiFi::gotResponse && Control2WiFi::responseType == START_CLIENT_TCP){
82  if(Control2WiFi::data[0]){ // client started correctly
83  Control2WiFi::state[_sock] = ESTABLISHED;
84  return 1;
85  }
86  break;
87  }
88  }
89  }
90  else{
91  Serial.println("No Socket available");
92  return 0;
93  }
94 
95  return 0;
96 }

137 int WiFiClient::read()
138 {
139  if(_internalBufSz > 0){
140  uint8_t ret = _internalBuf[_internalBufPtr++];
141  _internalBufSz--;
142 
143  return ret;
144  }
145 
146  if (available() <= 0)
147  return -1;
148 
149  Control2WiFi::handleEvents();
150 
151  Control2WiFi::gotResponse = false;
152  Control2WiFi::responseType = NONE;
153 
154  uint16_t sz = min(_sockDataSz, 25);
155 
156  if(!SocketDrv::getDataBuf(_sock, sz)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
157  // launch interrupt management function, then try to send request again
158  uint32_t start = millis();
159  while(millis() - start < 10)
160  Control2WiFi::handleEvents();
161  if(!SocketDrv::getDataBuf(_sock, sz)) // exit if another error occurs
162  return -1;
163  }
164  // Poll flags until we got a response or timeout occurs
165  uint32_t start = millis();
166 
167  while(((millis() - start) < 4000)){
168  Control2WiFi::handleEvents();
169  if(Control2WiFi::gotResponse && Control2WiFi::responseType == GET_DATABUF_TCP){
170  _internalBufPtr = 0;
171  _internalBufSz = (int8_t)Control2WiFi::dataLen;
172  memset(_internalBuf, 0, 25);
173  if(_internalBufSz <= 0){
174  _sockDataSz = 0;
175  _internalBufSz = 0;
176 
177  return -1;
178  }
179  memcpy(_internalBuf, (uint8_t *)Control2WiFi::data, _internalBufSz);
180 
181  uint8_t ret = _internalBuf[_internalBufPtr++];
182  _sockDataSz -= _internalBufSz;
183  _internalBufSz--;
184 
185  return ret;
186  }
187  }
188  return -1;
189 }

191 int WiFiClient::read(uint8_t* buf, size_t size) {
192  Control2WiFi::handleEvents();
193  Control2WiFi::gotResponse = false;
194  Control2WiFi::responseType = NONE;
195 
196  if(!SocketDrv::getDataBuf(_sock, size)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
197  // launch interrupt management function, then try to send request again
198  uint32_t start = millis();
199  while(millis() - start < 10)
200  Control2WiFi::handleEvents();
201  if(!SocketDrv::getDataBuf(_sock, size)) // exit if another error occurs
202  return -1;
203  }
204 
205  // Poll flags until we got a response or timeout occurs
206  uint32_t start = millis();
207  while(((millis() - start) < GENERAL_TIMEOUT)){
208  Control2WiFi::handleEvents();
209  if(Control2WiFi::gotResponse && Control2WiFi::responseType == GET_DATABUF_TCP){
210  uint16_t receivedBytes;
211  uint8_t chunkSize = 0;
212  int32_t totalLen = Control2WiFi::dataLen;
213 
214  if(totalLen <= 0){ // No data was read. Maybe an error occurred. Restore available flag and exit
215  Control2WiFi::client_data[_sock] = 0;
216  _sockDataSz = 0;
217  return -1;
218  }
219 
220  if(totalLen < 26)
221  receivedBytes = totalLen;
222  else
223  receivedBytes = 26;
224 
225  memcpy(buf, (uint8_t *)Control2WiFi::data, receivedBytes);
226 
227  while(receivedBytes < totalLen && (millis() - start) < 15000){ // wait 15 seconds at maximum for receiving the whole message. Otherwise consider that an error occurred
228  chunkSize = Control2WiFi::getAvailableData();
229  if(chunkSize > 0){
230  memcpy(&buf[receivedBytes], (void *)Control2WiFi::data, chunkSize);
231  receivedBytes += chunkSize;
232  }
233  }
234  Control2WiFi::client_data[_sock] -= receivedBytes;
235  _sockDataSz -= receivedBytes;
236  return receivedBytes;
237  }
238  }
239 
240  Control2WiFi::client_data[_sock] = 0;
241  _sockDataSz = 0;
242  return -1;
243 }
