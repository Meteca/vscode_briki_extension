24 void communicationEvtCb(uint8_t* pkt){
25  if (pkt != NULL) { // data was received during a write. copy data into commonDrv buffer before to decode them
26  memcpy(communication._rxBuf, pkt, BUF_SIZE);
27  } else {
28  // read the data out from the communication interface and put them into a buffer
29  uint8_t temp[BUF_SIZE];
30  uint8_t ret = communication._interface.readDataISR(temp);
31 
32  // decode the BUF_SIZE bytes long packet
33  if (ret == 0){
34  return;
35  }
36 
37  memcpy(communication._rxBuf, temp, BUF_SIZE);
38  }
39  uint8_t cmd = communication._rxBuf[1] - 0x80;
40 
41  if(communication._endReceived == false)
42  cmd = communication._currentCmd; // retrieve the command we are processing
43 
44  // check if command belongs to the user-specified ones
45  if(cmd == CUSTOM_CMD_CODE){
46  if(communication._cmdIndex){
47  uint8_t customCmd;
48  if(communication._endReceived == false)
49  customCmd = communication._customCurrentCmd;
50  else
51  customCmd = communication._rxBuf[6];
52  for(uint8_t i = 0; i < communication._cmdIndex; i++){
53  if(customCmd == communication._customCmdCode[i]){
54  if(communication._endReceived){ // new command
55  communication._totPktLen = (uint32_t)(communication._rxBuf[2] + (communication._rxBuf[3] << 8) + (communication._rxBuf[4] << 16) + (communication._rxBuf[5] << 24));
56  uint8_t copyLen;
57  if(communication._totPktLen <= 32){ // data are contained in a single packet
58  copyLen = communication._totPktLen - 8;
59  communication._endReceived = true;
60  // control the packet footer.
61  if(communication._rxBuf[communication._totPktLen - 1] != END_CMD){
62  return; // error
63  }
64 
65  // copy data
66  if(communication._custom_ext_buff[i] != NULL)
67  memcpy(&communication._custom_ext_buff[i][0], &communication._rxBuf[7], copyLen);
68  communication._custom_buff_index = copyLen;
69 
70  // call user defined callback
71  if(communication._userCallbacks[i] != NULL)
72  communication._userCallbacks[i](communication._custom_buff_index);
73 
74  communication._custom_buff_index = 0;
75  return;
76 
77  }
78  else{
79  copyLen = BUF_SIZE - 7; // buffer size - header
80  communication._endReceived = false;
81  communication._currentCmd = cmd;
82  communication._customCurrentCmd = customCmd;
83 
84  // copy the first chunk
85  if(communication._custom_ext_buff[i] != NULL)
86  memcpy(&communication._custom_ext_buff[i][0], &communication._rxBuf[7], copyLen);
87  communication._custom_buff_index = copyLen;
88  return;
89  }
90  }
91  else{ // command continuation
92  if((communication._custom_buff_index + BUF_SIZE) >= (communication._totPktLen - 8)){ // last chunk
93  uint8_t copyLen = BUF_SIZE - ((communication._custom_buff_index + BUF_SIZE) - (communication._totPktLen - 8));
94  if(communication._custom_ext_buff[i] != NULL)
95  memcpy(&communication._custom_ext_buff[i][communication._custom_buff_index], communication._rxBuf, copyLen);
96  communication._endReceived = true;
97  communication._custom_buff_index += copyLen;
98 
99  communication._totPktLen = 0;
100  // call user defined callback
101  if(communication._userCallbacks[i] != NULL)
102  communication._userCallbacks[i](communication._custom_buff_index);
103  communication._custom_buff_index = 0;
104  communication._currentCmd = 0;
105  return;
106  }
107  else{ // intermediate chunk
108  if(communication._custom_ext_buff[i] != NULL)
109  memcpy(&communication._custom_ext_buff[i][communication._custom_buff_index], communication._rxBuf, BUF_SIZE);
110  communication._custom_buff_index += BUF_SIZE;
111  return;
112  }
113  }
114  break;
115  }
116  }
117  }
118  return;
119  }
120 
121  // check all other commands
122  switch(cmd){
123  case GPIO_RESPONSE:
124  communication._responseType = GPIO_RESPONSE;
125  break;
126 
127  case GPIO_REQUEST:
128  communication._moveGpio();
129  break;
130 
131  // process PUT_CUSTOM_DATA command sent from the other micro
132  case PUT_CUSTOM_DATA:
133  if(communication._endReceived){ // new command
134  communication._totPktLen = (uint32_t)(communication._rxBuf[2] + (communication._rxBuf[3] << 8) + (communication._rxBuf[4] << 16) + (communication._rxBuf[5] << 24));
135  if(communication._user_ext_buff == NULL){ // create a safe place where storing data
136  // note: what follows is not a safe instruction for a microcontroller and should be avoided by using initExternalBuffer function.
137  communication._user_ext_buff = (uint8_t*)malloc(communication._totPktLen);
138  }
139  uint8_t copyLen;
140  if(communication._totPktLen <= 32){ // data are contained in a single packet
141  copyLen = communication._totPktLen - 7;
142  communication._endReceived = true;
143  // control the packet footer.
144  if(communication._rxBuf[communication._totPktLen - 1] != END_CMD){
145  return; // error
146  }
147 
148  // copy data
149  if(communication._user_ext_buff != NULL)
150  memcpy(communication._user_ext_buff, &communication._rxBuf[6], copyLen);
151  communication._user_ext_buff_sz = copyLen;
152 
153  return;
154 
155  }
156  else{
157  copyLen = BUF_SIZE - 6; // buffer size - header
158  communication._endReceived = false;
159  communication._currentCmd = cmd;
160 
161  // copy the first chunk
162  if(communication._user_ext_buff != NULL)
163  memcpy(communication._user_ext_buff, &communication._rxBuf[6], copyLen);
164  communication._user_ext_buff_sz = copyLen;
165  return;
166  }
167  }
168  else{ // command continuation
169  if((communication._user_ext_buff_sz + BUF_SIZE) >= (communication._totPktLen - 7)){ // last chunk
170  uint8_t copyLen = BUF_SIZE - ((communication._user_ext_buff_sz + BUF_SIZE) - (communication._totPktLen - 7));
171  if(communication._user_ext_buff != NULL)
172  memcpy(&communication._user_ext_buff[communication._user_ext_buff_sz], communication._rxBuf, copyLen);
173  communication._endReceived = true;
174  communication._user_ext_buff_sz += copyLen;
175 
176  communication._totPktLen = 0;
177 
178  communication._currentCmd = 0;
179  return;
180  }
181  else{ // intermediate chunk
182  if(communication._user_ext_buff != NULL)
183  memcpy(&communication._user_ext_buff[communication._user_ext_buff_sz], communication._rxBuf, BUF_SIZE);
184  communication._user_ext_buff_sz += BUF_SIZE;
185  return;
186  }
187  }
188  break;
189 
190  default:
191  // call user defined handlers
192  communication._responseType = NONE;
193  for(uint8_t i = 0; i < EXT_HANDLES; i++)
194  if(communication._handleExt[i] != NULL)
195  communication._handleExt[i](communication._rxBuf);
196 
197  break;
198  }

207 bool Communication::registerCommand(uint8_t cmdCode, ext_callback function, uint8_t* buffer){
208  handleCommEvents();
209 
210  if(_cmdIndex == CUSTOM_CMD_NR)
211  return false; // do not accept more than CUSTOM_CMD_NR commands
212 
213  for(uint8_t i = 0; i < _cmdIndex; i++){
214  if(_customCmdCode[i] == cmdCode)
215  return false; // do not accept an already assigned command code
216  }
217 
218  if(buffer != NULL)
219  _custom_ext_buff[_cmdIndex] = buffer;
220 
221  _customCmdCode[_cmdIndex] = cmdCode;
222  _userCallbacks[_cmdIndex] = function;
223  _cmdIndex++;
224  return true;
225 }

227 void Communication::sendCommand(uint8_t cmdCode, uint8_t* data, uint32_t len){
228  handleCommEvents();
229  bool existing = false;
230 
231  // check if cmdCode is an already registered one
232  for(uint8_t i = 0; i < _cmdIndex; i++){
233  if(_customCmdCode[i] == cmdCode){
234  existing = true;
235  break;
236  }
237  }
238  if(!existing) // do not send a command that has not been previously registered
239  return;
240 
241  communication.sendDataPkt(CUSTOM_CMD_CODE);
242  communication.appendByte(cmdCode);
243  communication.appendBuffer(data, len);

298 * Process GPIO_REQUEST command by actuating the selected operation on the specified pin
299 */
300 void Communication::_moveGpio(){
301  bool rep = false;
302  uint8_t* dataBuf = &_rxBuf[5]; // skip packet header
303  uint8_t op = dataBuf[0];
304  uint8_t pin = dataBuf[2];
305  uint8_t mode;
306  uint32_t val = 0;
307  uint32_t result = 0;
308 
309  switch(op){
310  case PIN_MODE: // pinMode command
311  mode = dataBuf[4];
312  pinMode(pin, mode);
313  rep = false; // no reply needed
314  break;
315  case WRITE_DIGITAL: // digitalWrite command
316  val = dataBuf[4] + (dataBuf[5] << 8) + (dataBuf[6] << 16) + (dataBuf[7] << 24);
317  digitalWrite(pin, val);
318  rep = false;
319  break;
320  case WRITE_ANALOG: // analogWrite command
321  val = dataBuf[4] + (dataBuf[5] << 8) + (dataBuf[6] << 16) + (dataBuf[7] << 24);
322  analogWrite(pin, val);
323 
324  rep = false;
325  break;
326  case READ_DIGITAL: // digitalRead command
327  result = digitalRead(pin);
328  rep = true;
329  break;
330  case READ_ANALOG: // analogRead command
331  result = analogRead(pin);
332  rep = true;
333  break;
334  default:
335  rep = false;
336  break;
337  }
338 
339  if(rep == true){
340  //send reply
341  _interface.sendCmd(GPIO_RESPONSE, PARAM_NUMS_1);
342  _interface.sendParam((uint8_t*)&result, 4, LAST_PARAM);
343  }
344  return;
345 }

419 * Activate digitalRead or analogRead function on one of the pin connected to the companion chip
420 *
421 * param op: a code representing the operation to be performed (READ_DIGITAL or READ_ANALOG)
422 * param pin: pin on which operation has to take effect

426 extern "C" int32_t sendGPIORead(uint8_t op, uint8_t pin){
427  communication.handleCommEvents();
428 
429  communication.clearResponseType();
430  communication.sendCmdPkt(GPIO_REQUEST, PARAM_NUMS_2);
431  communication.sendParam(&op, 1, NO_LAST_PARAM);
432  communication.sendParam(&pin, 1, LAST_PARAM);
433 
434  // Poll response for the next 2 seconds
435  uint32_t start = millis();
436  while (((millis() - start) < 2000)) {
437  communication.handleCommEvents();
438  if (communication.getResponseType() == GPIO_RESPONSE) {
439  uint8_t* dataBuf = communication.getRxBuffer();
440  dataBuf += 5; // skip packet header
441  uint32_t result = dataBuf[0] + (dataBuf[1] << 8) + (dataBuf[2] << 16) + (dataBuf[3] << 24);
442  return result;
443  }
444  }
445  return -1;
