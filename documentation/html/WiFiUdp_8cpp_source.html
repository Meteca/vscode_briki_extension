38 /* Start WiFiUDP socket, listening at local port PORT */
39 uint8_t WiFiUDP::begin(uint16_t port) {
40 
41  uint8_t sock = Control2WiFi::getSocket();
42  if (sock != NO_SOCKET_AVAIL)
43  {
44  if(!SocketDrv::startServer(port, sock, UDP_MODE)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
45  // launch interrupt management function, then try to send request again
46  uint32_t start = millis();
47  while(millis() - start < 10)
48  Control2WiFi::handleEvents();
49  if(!SocketDrv::startServer(port, sock, UDP_MODE)) // exit if another error occurs
50  return 0;
51  }
52 
53  // Poll response for the next 10 seconds
54  uint32_t start = millis();
55  while(((millis() - start) < GENERAL_TIMEOUT)){
56  Control2WiFi::handleEvents();
57  if(Control2WiFi::gotResponse && Control2WiFi::responseType == START_SERVER_TCP){
58  if(Control2WiFi::data[0] != 0){
59  Control2WiFi::server_port[sock] = port;
60  _sock = sock;
61  _port = port;
62  return 1;
63  }
64  }
65  }
66  }
67  return 0;

75  if(_sock != NO_SOCKET_AVAIL){

229 int WiFiUDP::read()
230 {
231  if(_internalBufSzUdp > 0){
232  uint8_t ret = _internalBufUdp[_internalBufPtrUdp++];
233  _internalBufSzUdp--;
234 
235  return ret;
236  }
237  if (available() <= 0)
238  return -1;
239 
240 
241 
242  Control2WiFi::handleEvents();
243 
244  Control2WiFi::gotResponse = false;
245  Control2WiFi::responseType = NONE;
246 
247  uint16_t sz = min(_sockDataSzUdp, 25);
248 
249  if(!SocketDrv::getDataBuf(_sock, sz)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
250  // launch interrupt management function, then try to send request again
251  uint32_t start = millis();
252  while(millis() - start < 10)
253  Control2WiFi::handleEvents();
254  if(!SocketDrv::getDataBuf(_sock, sz)) // exit if another error occurs
255  return -1;
256  }
257 
258  // Poll flags until we got a response or timeout occurs
259  uint32_t start = millis();
260  while(((millis() - start) < GENERAL_TIMEOUT)){
261  Control2WiFi::handleEvents();
262  if(Control2WiFi::gotResponse && Control2WiFi::responseType == GET_DATABUF_TCP){
263  _internalBufPtrUdp = 0;
264  _internalBufSzUdp = (int16_t)Control2WiFi::dataLen;
265  memset(_internalBufUdp, 0, 25);
266  if(_internalBufSzUdp <= 0){
267  _sockDataSzUdp = 0;
268  _internalBufSzUdp = 0;
269 
270  return -1;
271  }
272  memcpy(_internalBufUdp, (uint8_t *)Control2WiFi::data, _internalBufSzUdp);
273 
274  uint8_t ret = _internalBufUdp[_internalBufPtrUdp++];
275  _sockDataSzUdp -= _internalBufSzUdp;
276  _internalBufSzUdp--;
277 
278  return ret;
279 
280  }
281  }
282 
283  return -1;

286 int WiFiUDP::read(unsigned char* buffer, size_t len)
287 {
288  Control2WiFi::handleEvents();
289 
290  Control2WiFi::gotResponse = false;
291  Control2WiFi::responseType = NONE;
292 
293 
294  if(!SocketDrv::getDataBuf(_sock, len)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
295  // launch interrupt management function, then try to send request again
296  uint32_t start = millis();
297  while(millis() - start < 10)
298  Control2WiFi::handleEvents();
299  if(!SocketDrv::getDataBuf(_sock, len)) // exit if another error occurs
300  return -1;
301  }
302 
303  // Poll flags until we got a response or timeout occurs
304  uint32_t start = millis();
305  while(((millis() - start) < GENERAL_TIMEOUT)){
306  Control2WiFi::handleEvents();
307  if(Control2WiFi::gotResponse && Control2WiFi::responseType == GET_DATABUF_TCP){
308 
309  uint16_t receivedBytes;
310  uint8_t chunkSize = 0;
311  uint32_t totalLen = Control2WiFi::dataLen;
312 
313  if(totalLen <= 0){ // No data was read. Maybe an error occurred. Restore available flag and exit
314  Control2WiFi::client_data[_sock] = 0;
315  _sockDataSzUdp = 0;
316  return -1;
317  }
318 
319  if(totalLen < 26)
320  receivedBytes = totalLen;
321  else
322  receivedBytes = 26;
323 
324  memcpy(buffer, (uint8_t *)Control2WiFi::data, receivedBytes);
325 
326  while(receivedBytes < totalLen && (millis() - start) < 15000){ // wait 15 seconds at maximum for receiving the whole message. Otherwise consider that an error occurred
327  chunkSize = Control2WiFi::getAvailableData();
328  if(chunkSize > 0){
329  memcpy(&buffer[receivedBytes], (void *)Control2WiFi::data, chunkSize);
330  receivedBytes += chunkSize;
331  }
332  }
333  Control2WiFi::client_data[_sock] -= receivedBytes;
334  _sockDataSzUdp -= receivedBytes;
335  return receivedBytes;
336  }
337  }
338  Control2WiFi::client_data[_sock] = 0;
339  _sockDataSzUdp = 0;
340  return -1;
341 }
