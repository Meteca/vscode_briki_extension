33 void WiFiServer::begin()
34 {
35  Control2WiFi::handleEvents();
36 
37  Control2WiFi::gotResponse = false;
38  Control2WiFi::responseType = 0x00;
39 
40  uint8_t _sock = Control2WiFi::getSocket();
41 
42  if (_sock != NO_SOCKET_AVAIL) {
43  if(!SocketDrv::startServer(_port, _sock)){ // packet has not been sent. Maybe an interrupt occurred in the meantime
44  // launch interrupt management function, then try to send request again
45  uint32_t start = millis();
46  while(millis() - start < 10)
47  Control2WiFi::handleEvents();
48  if(!SocketDrv::startServer(_port, _sock)){ // exit if another error occurs
49  Serial.println("Error while starting server");
50  return;
51  }
52  }
53 
54  // Poll response for the next 10 seconds. If nothing happens print an error
55  uint32_t start = millis();
56  while(((millis() - start) < GENERAL_TIMEOUT)){
57  Control2WiFi::handleEvents();
58  if(Control2WiFi::gotResponse && Control2WiFi::responseType == START_SERVER_TCP){
59  if(Control2WiFi::data[0] != 0){
60  Control2WiFi::server_port[_sock] = _port;
61  return;
62  }
63  break;
64  }
65  }
66  Serial.println("Error while starting server");
67  }
68 }

70 WiFiClient WiFiServer::available()
71 {
72  Control2WiFi::handleEvents();
73  // search for a socket with a client request
74  for(uint8_t i = 0; i < MAX_SOCK_NUM; i++){
75  if(Control2WiFi::state[i] == ESTABLISHED){ // there is a client request
76  if(Control2WiFi::server_port[i] == _port){ // request refers to our socket
77  WiFiClient client(i);
78  // restore flag in order to be able to receive other clients on that socket
79  return client;
80  }
81  }
82  }
83  return WiFiClient(255);
84 }

91 size_t WiFiServer::write(const uint8_t *buffer, size_t size)
92 {
93  size_t n = 0;
94 
95  Control2WiFi::handleEvents();
96 
97  for (int sock = 0; sock < MAX_SOCK_NUM; sock++) {
98  if (Control2WiFi::server_port[sock] != 0) {
99  WiFiClient client(sock);
100 
101  if (Control2WiFi::server_port[sock] == _port && client.status() == ESTABLISHED) {
102  n+=client.write(buffer, size);
103  }
104  }
105  }
106  return n;
107 }
